620  inline void PROFILER_BEGIN( profilerQuery query )
621  {
622  const uint32_t evt = query*2;
623  const uint32_t flag = (1 << query);
624 
625  CUDA(cudaEventRecord(mEventsGPU[evt], mStream));
626  timestamp(&mEventsCPU[evt]);
627 
628  mProfilerQueriesUsed |= flag;
629  mProfilerQueriesDone &= ~flag;
630  }
631 
635  inline void PROFILER_END( profilerQuery query )
636  {
637  const uint32_t evt = query*2+1;
638 
639  CUDA(cudaEventRecord(mEventsGPU[evt]));
640  timestamp(&mEventsCPU[evt]);
641  timespec cpuTime;
642  timeDiff(mEventsCPU[evt-1], mEventsCPU[evt], &cpuTime);
643  mProfilerTimes[query].x = timeFloat(cpuTime);
644 
645  if( mEnableProfiler && query == PROFILER_NETWORK )
646  {
647  LogVerbose(LOG_TRT "layer network time - %f ms\n", gProfiler.timingAccumulator);
648  gProfiler.timingAccumulator = 0.0f;
649  LogWarning(LOG_TRT "note -- when processing a single image, run 'sudo jetson_clocks' before\n"
650  " to disable DVFS for more accurate profiling/timing measurements\n");
651  }
652  }
653 
657  inline bool PROFILER_QUERY( profilerQuery query )
658  {
659  const uint32_t flag = (1 << query);
660 
661  if( query == PROFILER_TOTAL )
662  {
663  mProfilerTimes[PROFILER_TOTAL].x = 0.0f;
664  mProfilerTimes[PROFILER_TOTAL].y = 0.0f;
665 
666  for( uint32_t n=0; n < PROFILER_TOTAL; n++ )
667  {
668  if( PROFILER_QUERY((profilerQuery)n) )
669  {
670  mProfilerTimes[PROFILER_TOTAL].x += mProfilerTimes[n].x;
671  mProfilerTimes[PROFILER_TOTAL].y += mProfilerTimes[n].y;
672  }
673  }
674 
675  return true;
676  }
677  else if( mProfilerQueriesUsed & flag )
678  {
679  if( !(mProfilerQueriesDone & flag) )
680  {
681  const uint32_t evt = query*2;
682  float cuda_time = 0.0f;
683  CUDA(cudaEventElapsedTime(&cuda_time, mEventsGPU[evt], mEventsGPU[evt+1]));
684  mProfilerTimes[query].y = cuda_time;
685  mProfilerQueriesDone |= flag;
686  //mProfilerQueriesUsed &= ~flag;
687  }
688 
689  return true;
690  }
691 
692  return false;
693  }
